module ListMonad

: sig

Monad type which represents a list of results.

#
type 'a m = 'a list
#
val bind : 'a m -> ('a -> 'b m) -> 'b m

bind x f applies f to a list of results, returning a list of results.

#
val (>>=) : 'a m -> ('a -> 'b m) -> 'b m
#
val return : 'a -> 'a m

return x is the left and right unit of bind.

end
